\documentclass[
  fontsize=11pt,
  paper=a4,
  twoside,
  open=right,
  parskip=half,
  numbers=noenddot,
]{scrbook}

% --- LuaLaTeX-Schrift-/Typografie-Setup ---
\usepackage{fontspec}
\usepackage{microtype}
\usepackage[ngerman]{babel}
\usepackage{csquotes}   % empfohlener Begleiter für babel + biblatex

% Backticks (`) werden in verbatim-Umgebungen als solche dargestellt
% (siehe https://tex.stackexchange.com/a/341057/9075)
\usepackage{upquote}

% --- Markdown als Texteingabe, großzügiger Funktionsumfang ---
% Die Optionsnamen folgen dem markdown-Paket v3.x.
\usepackage[
  citations,          % [@key] Zitationssyntax
  hybrid,             % rohes LaTeX durchreichen (z. B. \zcref im Manuskript)
  notes,              % [^label] referenzierte Fußnoten
  inlineNotes,        % ^[Inline-Fußnotentext]
  pipeTables,         % | a | b | Tabellen
  tableCaptions,      % : Beschriftungstext unter einer Tabelle
  tableAttributes,    % {#id} Bezeichner an einer Tabellenbeschriftung
  hashEnumerators,    % 1) 2) 3) Aufzählungen
  taskLists,          % - [ ] / - [x] Checkbox-Listen
  headerAttributes,   % {#id} Bezeichner an einer Überschrift
  linkAttributes,     % {#id} Attribute an Links/Bildern
  relativeReferences, % <#id> zum Referenzieren einer via {#id} definierten Id
]{markdown}

% --- Literatur: Autor-Jahr über biblatex/biber ---
\usepackage[
  style=authoryear,
  backend=biber,
  autolang=hyphen,
  sorting=nyt,
]{biblatex}
\addbibresource{bibliography.bib}

% --- Abbildungen + intelligente Querverweise ---
\usepackage{graphicx}
\usepackage{zref-clever}
\zcsetup{cap = true}    % Typnamen groß schreiben: "Kapitel", "Abbildung", "Tabelle"

% Wie die Marken funktionieren (Antwort auf "Können wir nicht die Markdown-Marken nutzen?"):
% Das markdown-Paket macht aus `{#id}` (an Überschriften, Bildern und Tabellen) ein
% *reguläres* \label{id}. Der `labelhook` von zref-clever (standardmäßig an) macht
% jedes \label zusätzlich zu einer zref-Marke, sodass \zcref{id} direkt mit
% Markdowns eigenen Bezeichnern funktioniert -- ohne rohes \zlabel oder Handarbeit.

% --- Alles anklickbar: hyperref ZULETZT laden -----------------------------
% Macht Inhaltsverzeichnis, \zcref, native <#id>-Verweise, [@Zitate] und Fußnoten
% anklickbar. zref-clever lädt zref-hyperref selbst, sobald es hyperref sieht, sodass
% \zcref-Links kein weiteres Setup brauchen. `hidelinks` hält Links anklickbar, ohne
% farbige Kästen (statt dessen `colorlinks=true` für sichtbare farbige Links am Bildschirm).
\usepackage[hidelinks, unicode, pdfusetitle]{hyperref}

\title{Mein Buch}
\author{Martha Mildenhall}
\date{\today}

\begin{document}

\frontmatter
\maketitle
\tableofcontents

\mainmatter

% Der Fließtext liegt in einer separaten, reinen Markdown-Datei -- so sieht der
% Editor Markdown (kein LaTeX) und die Rechtschreib-/Stilprüfer (Vale, textlint)
% prüfen ihn direkt. `hybrid` (oben gesetzt) lässt darin weiterhin rohes LaTeX
% wie \zcref durch.
\markdownInput{manuscript-de.md}

\backmatter

\printbibliography

\end{document}
